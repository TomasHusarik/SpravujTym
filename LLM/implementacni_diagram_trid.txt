class User {
  +_id : ObjectId
  +name : String
  +email : String
  +mobile : String
  +age : Number
  +sex : String
  +address : String
  +birthDate : Date
  +active : Boolean
  +createdAt : Date
}

class League {
  +_id : ObjectId
  +name : String
  +season : String
  +category : String
}

class Team {
  +_id : ObjectId
  +name : String
  +shortName : String
  +homeCity : String
  +foundedYear : Number
}

class Squad {
  +_id : ObjectId
  +name : String
  +teamId : ObjectId
  +leagueId : ObjectId
}

class Membership {
  +_id : ObjectId
  +userId : ObjectId
  +squadId : ObjectId
  +role : String
  +active : Boolean
  +startDate : Date
  +endDate : Date
}

class TeamEvent {
  +_id : ObjectId
  +title : String
  +type : String
  +opponent : String
  +startTime : Date
  +endTime : Date
  +squadId : ObjectId
  +venueId : ObjectId
}

class EventParticipation {
  +_id : ObjectId
  +eventId : ObjectId
  +userId : ObjectId
  +status : String
  +note : String
  +createdAt : Date
}

class Venue {
  +_id : ObjectId
  +name : String
  +address : String
  +city : String
  +capacity : Number
}

class Announcement {
  +_id : ObjectId
  +title : String
  +content : String
  +createdAt : Date
  +visibility : String
  +type : String
  +pinned : Boolean
  +teamId : ObjectId
  +authorId : ObjectId
}

class Payment {
  +_id : ObjectId
  +userId : ObjectId
  +amount : Number
  +status : String
  +dueDate : Date
  +type : String
}
